# Function to transfer a single table
transfer_table <- function(table_name, source_conn, target_conn) {
  cat("\n📤 Transferring table:", table_name, "\n")
  
  # Read data from source
  source_data <- tryCatch({
    dbReadTable(source_conn, table_name)
  }, error = function(e) {
    cat("  ⚠️  Error reading table:", e$message, "\n")
    return(NULL)
  })
  
  if (is.null(source_data) || nrow(source_data) == 0) {
    cat("  ℹ️  Table is empty or could not be read, skipping...\n")
    return(FALSE)
  }
  
  cat("  📊 Read", nrow(source_data), "rows\n")
  
  # Check if table exists in target
  if (dbExistsTable(target_conn, table_name)) {
    cat("  ⚠️  Table already exists in target, dropping...\n")
    dbRemoveTable(target_conn, table_name)
  }
  
  # Write to target
  tryCatch({
    dbWriteTable(target_conn, table_name, source_data, overwrite = TRUE)
    cat("  ✅ Successfully transferred", nrow(source_data), "rows\n")
    return(TRUE)
  }, error = function(e) {
    cat("  ❌ Error writing to target:", e$message, "\n")
    return(FALSE)
  })
}
